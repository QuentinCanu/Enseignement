\begin{parts}
\part Justifier que $f$ est dérivable sur $\R$.
\part En déduire l'expression de $f'(x)$ pour tout $x \in \R$.
\part Construire le tableau de signes de $f'$.
\part En déduire le tableau de variations de $f$.
\end{parts}

\begin{exercize}[Fractions]
Effectuer les calculs suivants. On donnera le résultat sous forme de fraction réduite.
\begin{alphaquestions}
\begin{minipage}{0.45\textwidth}
\item $\dfrac{7}{12} - \dfrac{4}{9}$
\item $\dfrac{9}{7} + \dfrac{1}{35}$
\item $\dfrac{77}{-45} \times \dfrac{-15}{-44}$
\end{minipage}
\hfill
\begin{minipage}{0.45\textwidth}
\item $\dfrac{-25}{-42} \times \dfrac{7}{50}$
\item $\dfrac{4}{5} \div \dfrac{7}{10}$
\item Rendre $\dfrac{2925}{2475}$ irréductible
\end{minipage}
\end{alphaquestions} 
\end{exercize}
\begin{exercize}[Puissances]
Effectuer les calculs suivants. On donnera le résultat sous forme de $a^n$, avec $a \in \R$ et $n \in \N$.
\begin{alphaquestions}
\begin{minipage}{0.45\textwidth}
\item $\dfrac{27^3}{3}$
\item $\dfrac{4^5}{2}$
\item $\dfrac{3^7 \times 3^6}{9^4} \times 3$
\item $\dfrac{2^2 \times 8}{2^6}$
\end{minipage}
\hfill
\begin{minipage}{0.45\textwidth}
\item $\dfrac{5 \times 5^7}{25}$
\item $\dfrac{2 \times 2^2}{4 \times 4}$
\item $\dfrac{8 \times 2}{4^5}$
\item $\dfrac{3^4 \times 9}{3^5 \times 3^6}$
\end{minipage}
\end{alphaquestions} 
\end{exercize}
\begin{exercize}[Pourcentages]
Répondre aux questions suivantes :
\begin{alphaquestions}
\item Donner ces fraction sous forme décimale et de pourcentage: $\dfrac{31}{1000}$; $\dfrac{46}{500}$; $\dfrac{91}{2500}$ \dots 
\item Un sac de $2340$ patates contient $20 \%$ de patates douces. Combien y a-t-il de patates douces ?
\item Lors d'un concert, il y a $135$ spectateurs de plus de $60$ ans, ce qui représente $18\%$ des spectateurs. Combien y a-t-il de spectateurs au total ?
\end{alphaquestions} 
\end{exercize}
\begin{exercize}[Évolutions]
Répondre aux questions suivantes :
\begin{alphaquestions}
\item Dire à quelles multiplications sont équivalentes les évolutions en pourcentage suivantes :
\begin{itemize}
\item \og Diminuer de $16\%$ \fg
\item \og Augmenter de $2\%$ \fg
\item \og Augmenter de $140\%$ \fg
\end{itemize}
\item Le prix d'un article subit une baisse de $14\%$ suivi d'une baisse de $76\%$. Donner le taux d'évolution global du prix.
\item Le nombre de commerciaux d'une entreprise a augmenté de $3\%$. Quelle évolution en pourcentage permettrait de retrouver le pourcentage de départ ?
\end{alphaquestions}
\end{exercize}

\begin{exercize}[Égalités]
Exprimer la variable demandée en fonction des autres à partir de l'égalité donnée :
\begin{alphaquestions}
\begin{minipage}{0.4\textwidth}
\item $b = ec + a$; exprimer $a$ en fonction de $b$, $e$ et $c$;
\item $w = \dfrac{r + u}{v - t}$; exprimer $t$ en fonction de $w$, $r$, $u$ et $v$;
\end{minipage}
\hfill
\begin{minipage}{0.4\textwidth}
\item $s = \dfrac{t + u}{r}$; exprimer $u$ en fonction de $s$, $t$ et $r$;
\item $B = \dfrac{F + C}{A - E}$; exprimer $A$ en fonction de $B$, $F$, $C$ et $E$.
\end{minipage}
\end{alphaquestions}
\end{exercize}

\end{alphaquestions}
\end{exercize}

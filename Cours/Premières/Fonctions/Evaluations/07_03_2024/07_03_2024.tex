\begin{center}
\begin{tikzpicture}
\draw[line width = 0.01cm] (-3.25,-3.25) grid[step=0.5] (3.25,3.25);
\draw[ultra thick,->] (-3.25,0) -- (3.25,0) node[right] {$x$};
\draw[ultra thick,->] (0,-3.25) -- (0,3.25) node[above] {$y$};
\draw[thick] (0.5,0.2) -- (0.5,-0.2) node[left] {$1$};
